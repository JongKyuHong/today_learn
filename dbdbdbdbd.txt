## 01. 테이블을 만들고, 열을 추가하기
create table test1(
    -> hello varchar(10),
    -> hi varchar(20),
    -> name varchar(30),
    -> number varchar(40)
    -> );

alter table test add(id int(10));

## 02. 열을 삭제하기

alter table test1 drop id;

## 03. 열을 변경한다

alter table test1 CHANGE hello hello2 varchar(10);
